400 QJsonChannelMessage QJsonChannelService::dispatch (const QJsonChannelMessage& request) {
401  Q_D (QJsonChannelService);
402  if (request.type () != QJsonChannelMessage::Request && request.type () != QJsonChannelMessage::Notification) {
403  return request.createErrorResponse (QJsonChannel::InvalidRequest, "invalid request");
404  }
405 
406  const QByteArray& method (methodName (request));
407  if (!d->invokableMethodHash.contains (method)) {
408  return request.createErrorResponse (QJsonChannel::MethodNotFound, "invalid method called");
409  }
410 
411  int idx = -1;
412  QVariantList arguments;
413  const QList<int>& indexes = d->invokableMethodHash.value (method);
414  const QJsonValue& params = request.params ();
415  QVarLengthArray<void*, 10> parameters;
416  QVariant returnValue;
417  QMetaType::Type returnType = QMetaType::Void;
418 
419  bool usingNamedParameters = params.isObject ();
420 
421  // iterate over candidates
422  foreach (int methodIndex, indexes) {
423  QJsonChannelServicePrivate::MethodInfo& info = d->methodInfoHash[methodIndex];
424  bool methodMatch = usingNamedParameters ? jsParameterCompare (params.toObject (), info) : jsParameterCompare (params.toArray (), info);
425 
426  if (methodMatch) {
427  idx = methodIndex;
428  arguments.reserve (info.parameters.size ());
429  returnType = static_cast<QMetaType::Type> (info.returnType);
430  returnValue = (returnType == QMetaType::Void) ? QVariant () : QVariant (returnType, Q_NULLPTR);
431  if (returnType == QMetaType::QVariant)
432  parameters.append (&returnValue);
433  else
434  parameters.append (returnValue.data ());
435 
436  for (int i = 0; i < info.parameters.size (); ++i) {
437  const QJsonChannelServicePrivate::ParameterInfo& parameterInfo = info.parameters.at (i);
438  QJsonValue incomingArgument = usingNamedParameters ? params.toObject ().value (parameterInfo.name) : params.toArray ().at (i);
439 
440  QVariant argument = convertArgument (incomingArgument, parameterInfo);
441  if (!argument.isValid ()) {
442  QString message = incomingArgument.isUndefined () ? QString ("failed to construct default object for '%1'").arg (parameterInfo.name)
443  : QString ("failed to convert from JSON for '%1'").arg (parameterInfo.name);
444  return request.createErrorResponse (QJsonChannel::InvalidParams, message);
445  }
446 
447  arguments.push_back (argument);
448  if (parameterInfo.type == QMetaType::QVariant)
449  parameters.append (static_cast<void*> (&arguments.last ()));
450  else
451  parameters.append (const_cast<void*> (arguments.last ().constData ()));
452  }
453 
454  // found a match
455  break;
456  }
457  }
458 
459  if (idx == -1) {
460  return request.createErrorResponse (QJsonChannel::InvalidParams, "invalid parameters");
461  }
462 
463  QJsonChannelServicePrivate::MethodInfo& info = d->methodInfoHash[idx];
464 
465  QObject* serviceObj = d->serviceObj;
466  bool success = serviceObj->qt_metacall (QMetaObject::InvokeMetaMethod, idx, parameters.data ()) < 0;
467  if (!success) {
468  QString message = QString ("dispatch for method '%1' failed").arg (method.constData ());
469  return request.createErrorResponse (QJsonChannel::InvalidRequest, message);
470  }
471 
472  if (info.hasOut) {
473  QJsonArray ret;
474  if (info.returnType != QMetaType::Void)
475  ret.append (QJsonChannelServicePrivate::convertReturnValue (returnValue));
476  for (int i = 0; i < info.parameters.size (); ++i)
477  if (info.parameters.at (i).out)
478  ret.append (QJsonChannelServicePrivate::convertReturnValue (arguments[i]));
479  if (ret.size () > 1)
480  return request.createResponse (ret);
481  return request.createResponse (ret.first ());
482  }
483
